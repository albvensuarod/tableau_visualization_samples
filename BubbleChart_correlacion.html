# Crear el grfico de burbujas con seaborn
plt.figure(figsize=(10, 8))
sns.scatterplot(x=df_corr.columns, y=df_corr.columns, size=0, legend=False)  # Crear el eje x e y

# Agregar burbujas con tamao proporcional a los valores de correlacin
for i in range(len(df_corr.columns)):
    for j in range(len(df_corr.columns)):
        corr_value = df_corr.iloc[i, j]
        size = 1000 * abs(corr_value)  # Tamao proporcional al valor absoluto de correlacin
        plt.scatter(df_corr.columns[i], df_corr.columns[j], s=size, alpha=0.6)  # Dibujar la burbuja
        plt.text(df_corr.columns[i], df_corr.columns[j], f'{corr_value:.2f}', ha='center', va='center')  # Agregar texto con el valor de correlacin

# Aadir etiquetas y ttulo
plt.xlabel('Variables')
plt.ylabel('Variables')
plt.title('Bubble Chart de Correlaciones')

# Mostrar el grfico
plt.xticks(rotation=45, ha='right')  # Rotar las etiquetas del eje x para mejor visibilidad
plt.yticks(rotation=0)
plt.show()
